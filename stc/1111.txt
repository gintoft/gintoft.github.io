int nod_OBICHNIY(int a, int b){
	int min = a<b ? a : b;
	int max = a>b ? a : b;
	for (int i=min;i>=1;i--){
		if (a%i==0 && b%i==0){
			return i;
		}
	}
	return -1;
}

int nod_ev(int a, int b) {
  int min = a<b ? a : b;
  int max = a>b ? a : b;
  if (min==0) {
    return max;
  } else {
    return nod_ev(min, max%min);
  }
  return -1;
}
